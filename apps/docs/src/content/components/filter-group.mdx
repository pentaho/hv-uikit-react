### Usage

Use the prop `filterContentProps.iconOnly` to display the trigger component as an icon
button (only available on the Pentaho theme).

```tsx live
import { useState } from "react";

export default function Demo() {
  const [iconOnly, setIconOnly] = useState(false);
  const [disabled, setDisabled] = useState(false);
  const [value, setValue] = useState<HvFilterGroupValue | undefined>([
    ["format1", "format2"],
    [],
    ["data2", "data8"],
  ]);

  return (
    <>
      <div className="flex flex-col gap-sm p-sm">
        <HvTypography variant="label">Applied filters</HvTypography>

        <div className="flex flex-wrap gap-xs">
          {value?.flat().map((filterId) => {
            const tagLabel = filters
              .flatMap((group) => group.data)
              .find((item) => item.id === filterId)?.name;
            return <HvTag key={filterId} label={tagLabel} />;
          })}
        </div>

        <div className="flex justify-between">
          <div className="w-200px">
            <HvFilterGroup
              aria-label="Main filter group"
              value={value}
              disabled={disabled}
              filters={filters}
              onChange={(_, values) => {
                setValue(values);
              }}
              filterContentProps={{
                iconOnly,
              }}
            />
          </div>
          <div className="flex flex-col gap-xs">
            <HvCheckBox
              label="Icon only"
              checked={iconOnly}
              onChange={(_, checked) => setIconOnly(checked)}
            />
            <HvCheckBox
              label="Disabled"
              checked={disabled}
              onChange={(_, checked) => setDisabled(checked)}
            />
          </div>
        </div>
      </div>
    </>
  );
}

const filters: HvFilterGroupProps["filters"] = [
```
