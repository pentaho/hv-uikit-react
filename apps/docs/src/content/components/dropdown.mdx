### Groups

Use the `separator` property on any data item to create groups of items within the dropdown.

```tsx live
<HvDropdown
  values={[
    {
      id: "import",
      label: "Import",
      icon: (
        <HvIconContainer>
          <div className="i-ph-download-simple" />
        </HvIconContainer>
      ),
    },
    {
      id: "export",
      label: "Export",
      icon: (
        <HvIconContainer>
          <div className="i-ph-export" />
        </HvIconContainer>
      ),
    },
    {
      id: "share",
      label: "Share",
      icon: (
        <HvIconContainer>
          <div className="i-ph-share-network" />
        </HvIconContainer>
      ),
      separator: true,
    },
    {
      id: "settings",
      label: "Settings",
      icon: (
        <HvIconContainer>
          <div className="i-ph-sliders" />
        </HvIconContainer>
      ),
    },
    {
      id: "filters",
      label: "Filters",
      icon: (
        <HvIconContainer>
          <div className="i-ph-funnel" />
        </HvIconContainer>
      ),
      separator: true,
    },
    {
      id: "delete",
      label: "Delete",
      icon: (
        <HvIconContainer>
          <div className="i-ph-trash" />
        </HvIconContainer>
      ),
    },
  ]}
/>
```

### Customized items

The items in the `values` prop can be can be customized, namely with an icon or custom content on the `label` prop. Each item can be individually disabled also.

```tsx live
<HvDropdown
  values={[
    { id: "save", label: "Save", icon: <Save /> },
    { id: "export", label: "Export", icon: <Open />, disabled: true },
    {
      id: "preview",
      label: <HvTypography variant="label">Preview</HvTypography>,
      icon: <Preview />,
    },
  ]}
/>
```
